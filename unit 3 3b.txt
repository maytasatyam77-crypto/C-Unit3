//3b: Pointer to Structure
//Program to Access Structure Members Using Pointer to Structure:
#include <stdio.h>
struct student {
    int roll;
    char name[30];
    float marks;
};

int main() {
    struct student s[100];
    struct student *ptr;
    int n, i;

    ptr = s;   

    printf("Enter number of students: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        printf("\nEnter details of student %d\n", i + 1);

        printf("Roll number: ");
        scanf("%d", &ptr[i].roll);

        printf("Name: ");
        scanf("%s", ptr[i].name);

        printf("Marks: ");
        scanf("%f", &ptr[i].marks);
    }
    
    printf("\n--- Student Records ---\n");
    for (i = 0; i < n; i++) {
        printf("\nStudent %d\n", i + 1);
        printf("Roll: %d\n", ptr[i].roll);
        printf("Name: %s\n", ptr[i].name);
        printf("Marks: %.2f\n", ptr[i].marks);
    }

    return 0;
}
