//3c: Passing Structure to Function Program to Pass Structure to Function Using Pointer (Call by Reference)
#include <stdio.h>
struct Student {
    int id;
    char name[20];
    float marks;
};

void display(struct Student *s);

int main() {
    struct Student s1 = {157, "Satyam", 85.5};

    display(&s1);

    return 0;
}
void display(struct Student *s) {
    printf("ID: %d\n", s->id);
    printf("Name: %s\n", s->name);
    printf("Marks: %.2f\n", s->marks);
}
